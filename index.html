import os import logging import requests from flask import Flask, redirect, request, render_template, jsonify, session, url_for from dotenv import load_dotenv import openai from urllib.parse import quote # Завантаження змінних середовища load_dotenv() # Налаштування логування logging.basicConfig(level=logging.INFO) logger = logging.getLogger(__name__) # Ініціалізація Flask додатку app = Flask(__name__) app.secret_key = os.getenv('SECRET_KEY', os.urandom(24)) # Конфігурація class Config: CLIENT_ID = os.getenv('CLIENT_ID') CLIENT_SECRET = os.getenv('CLIENT_SECRET') REDIRECT_URI = os.getenv('REDIRECT_URI', 'http://127.0.0.1:5000/callback') OPENAI_API_KEY = os.getenv('OPENAI_API_KEY') AUTH_URL = 'https://www.linkedin.com/oauth/v2/authorization' TOKEN_URL = 'https://www.linkedin.com/oauth/v2/accessToken' PROFILE_URL = 'https://api.linkedin.com/v2/me' SCOPES = 'r_liteprofile' config = Config() # Перевірка змінних середовища required_vars = ['CLIENT_ID', 'CLIENT_SECRET', 'OPENAI_API_KEY'] missing_vars = [var for var in required_vars if not getattr(config, var)] if missing_vars: raise ValueError(f"Відсутні змінні середовища: {', '.join(missing_vars)}") openai.api_key = config.OPENAI_API_KEY # Маршрути @app.route('/') def index(): return render_template('index.html') @app.route('/login') def login(): try: state = os.urandom(16).hex() session['oauth_state'] = state params = { 'response_type': 'code', 'client_id': config.CLIENT_ID, 'redirect_uri': config.REDIRECT_URI, 'scope': config.SCOPES, 'state': state } auth_url = f"{config.AUTH_URL}?{'&'.join(f'{k}={quote(v)}' for k, v in params.items())}" return redirect(auth_url) except Exception as e: logger.error(f'Помилка авторизації: {e}') return jsonify({'error': 'Authorization failed'}), 500 @app.route('/callback') def callback(): try: code = request.args.get('code') state = request.args.get('state') error = request.args.get('error') if error: return jsonify({'error': error, 'description': request.args.get('error_description')}), 400 if not code or state != session.get('oauth_state'): return jsonify({'error': 'Invalid authorization request'}), 400 access_token = get_access_token(code) if not access_token: return jsonify({'error': 'Failed to obtain access token'}), 500 profile_data = get_user_profile(access_token) if not profile_data: return jsonify({'error': 'Failed to fetch profile data'}), 500 session['access_token'] = access_token session['profile_data'] = profile_data return render_template('profile.html', profile=profile_data) except Exception as e: logger.error(f'Callback error: {e}') return jsonify({'error': 'Callback processing failed'}), 500 # Допоміжні функції def get_access_token(code): try: data = { 'grant_type': 'authorization_code', 'code': code, 'redirect_uri': config.REDIRECT_URI, 'client_id': config.CLIENT_ID, 'client_secret': config.CLIENT_SECRET, } response = requests.post( config.TOKEN_URL, data=data, headers={'Content-Type': 'application/x-www-form-urlencoded'}, timeout=10 ) response.raise_for_status() return response.json().get('access_token') except requests.exceptions.RequestException as e: logger.error(f'Error fetching access token: {e}') return None def get_user_profile(access_token): try: headers = { 'Authorization': f'Bearer {access_token}', 'Connection': 'Keep-Alive' } response = requests.get(config.PROFILE_URL, headers=headers, timeout=10) response.raise_for_status() return response.json() except requests.exceptions.RequestException as e: logger.error(f'Error fetching profile: {e}') return None # Обробники помилок @app.errorhandler(404) def page_not_found(e): return render_template('404.html'), 404 @app.errorhandler(500) def internal_server_error(e): return render_template('500.html'), 500 if __name__ == '__main__': app.run(debug=True, host='127.0.0.1', port=5000, threaded=True)
